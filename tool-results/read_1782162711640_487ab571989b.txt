    57→/** Parse comma-separated category string into array */
    58→function parseCategories(categoryStr: string): string[] {
    59→  if (!categoryStr) return []
    60→  return categoryStr.split(',').map(c => c.trim()).filter(Boolean)
    61→}

    72→  const [form, setForm] = useState<ChannelFormData>(emptyForm)
    73→  const [saving, setSaving] = useState(false)
    74→
    75→  // IPTV Import state

    81→  const [importing, setImporting] = useState(false)
    82→  const [iptvSearch, setIptvSearch] = useState('')
    83→
    84→  // File Import state
    85→  const [showFileImport, setShowFileImport] = useState(false)
    86→  const [fileImportLoading, setFileImportLoading] = useState(false)
    87→  const [fileImportResults, setFileImportResults] = useState<{ name: string; logo: string; group: string; url: string; language?: string; country?: string }[]>([])
    88→  const [selectedFileChannels, setSelectedFileChannels] = useState<Set<number>>(new Set())
    89→  const [fileImporting, setFileImporting] = useState(false)
    90→  const [fileImportSearch, setFileImportSearch] = useState('')

   100→

   104→  const [channelViewers, setChannelViewers] = useState<Record<string, number>>({})

   109→  const loadChannels = useCallback(async () => {

   111→      setLoading(true)
   112→      const data = await fetchChannels({
   113→        ...(filterCategory !== 'all' ? { category: filterCategory } : {}),
   114→        ...(searchQuery ? { search: searchQuery } : {}),
   115→        includeInactive: true,
   116→      })
   117→      setChannels(data)
   118→    } catch {
   119→      toast.error('Error', { description: 'Failed to load channels' })
   120→    } finally {
   121→      setLoading(false)
   122→    }
   123→  }, [filterCategory, searchQuery])
   124→
   125→  useEffect(() => {
   126→    loadChannels()
   127→  }, [loadChannels])
   128→
   129→  // Poll /api/admin/live-viewers every 5s to keep the "Live" column as close
   130→  // to real-time as possible. Sends the current list of channel ids so the
   131→  // server can return counts for exactly those channels (no demo data — a
   132→  // channel with no viewers shows 0). The poll stops when the component

   134→  useEffect(() => {

   137→
   138→    const fetchLiveViewers = async () => {

   141→          method: 'POST',
   142→          headers: { 'Content-Type': 'application/json' },
   143→          body: JSON.stringify({ channelIds: channels.map((c) => c.id) }),
   144→          signal: controller.signal,

   146→        if (!res.ok) return
   147→        const data = await res.json()
   148→        if (data && typeof data.channelViewers === 'object') {
   149→          setChannelViewers(data.channelViewers)
   150→        }
   151→      } catch {
   152→        // Silent — analytics must never break the admin panel.
   153→      }
   154→    }
   155→
   156→    fetchLiveViewers()
   157→    const timer = setInterval(fetchLiveViewers, 5_000)
   158→    return () => {
   159→      controller.abort()
   160→      clearInterval(timer)
   161→    }
   162→  }, [channels])
   163→
   164→  const handleSave = async () => {
   165→    if (!form.name.trim()) {
   166→      toast.error('Validation Error', { description: 'Channel name is required' })
   167→      return
   168→    }
   169→
   170→    setSaving(true)
   171→    try {
   172→      const data = {
   173→        name: form.name,
   174→        logo: form.logo,
   175→        category: form.categories.filter(Boolean).join(','),
   176→        streamType: form.streamType,
   177→        streamUrl: form.streamUrl,
   178→        githubM3uPath: form.githubM3uPath,
   179→        language: form.language,
   180→        country: form.country,
   181→        tags: form.tags,
   182→        isFeatured: form.isFeatured,
   183→        isActive: form.isActive,
   184→      }
   185→
   186→      if (editingId) {
   187→        await updateChannel(editingId, data)
   188→        toast.success('Channel Updated', { description: `${form.name} has been updated successfully` })

   209→      name: channel.name,

   213→      streamUrl: channel.streamUrl,
   214→      githubM3uPath: channel.githubM3uPath,
   215→      language: channel.language,
   216→      country: channel.country,
   217→      tags: channel.tags,
   218→      isFeatured: channel.isFeatured,

   239→      await deleteChannel(id)
   240→      toast.success('Channel Deleted', { description: 'Channel has been removed' })
   241→      setDeleteConfirm(null)
   242→      loadChannels()
   243→    } catch {
   244→      toast.error('Error', { description: 'Failed to delete channel' })
   245→    }
   246→  }
   247→
   248→  const handleBulkDelete = async () => {
   249→    if (selectedIds.size === 0) return
   250→    if (!confirm(`Are you sure you want to delete ${selectedIds.size} channel${selectedIds.size > 1 ? 's' : ''}? This cannot be undone.`)) return
   251→
   252→    setBulkDeleting(true)
   253→    let deleted = 0
   254→    let failed = 0
   255→    for (const id of selectedIds) {
   256→      try {
   257→        await deleteChannel(id)

   260→        failed++

   263→    setBulkDeleting(false)
   264→    setSelectedIds(new Set())

   266→    if (failed > 0) toast.error(`Failed to delete ${failed} channel${failed > 1 ? 's' : ''}`)
   267→    loadChannels()
   268→  }
   269→
   270→  const toggleSelect = (id: string) => {
   271→    setSelectedIds(prev => {
   272→      const next = new Set(prev)
   273→      if (next.has(id)) next.delete(id)
   274→      else next.add(id)
   275→      return next
   276→    })
   277→  }
   278→
   279→  const toggleSelectAll = () => {
   280→    if (selectedIds.size === channels.length) {
   281→      setSelectedIds(new Set())
   282→    } else {
   283→      setSelectedIds(new Set(channels.map(ch => ch.id)))
   284→    }
   285→  }
   286→
   287→  const handleToggleActive = async (channel: Channel) => {
   288→    try {
   289→      await updateChannel(channel.id, { isActive: !channel.isActive })
   290→      toast.success('Channel Updated', { description: `${channel.name} is now ${!channel.isActive ? 'active' : 'inactive'}` })
   291→      loadChannels()
   292→    } catch {
   293→      toast.error('Error', { description: 'Failed to update channel' })
   294→    }
   295→  }
   296→
   297→  const handleToggleFeatured = async (channel: Channel) => {
   298→    try {
   299→      await updateChannel(channel.id, { isFeatured: !channel.isFeatured })
   300→      toast.success('Channel Updated', { description: `${channel.name} featured status updated` })
   301→      loadChannels()
   302→    } catch {
   303→      toast.error('Error', { description: 'Failed to update channel' })
   304→    }
   305→  }
   306→
   307→  const handleIptvParse = async () => {
   308→    if (!iptvUrl.trim()) return
   309→    setIptvLoading(true)
   310→    try {
   311→      const result = await parseM3U(iptvUrl)
   312→      setIptvResults(result.channels)
   313→      setSelectedIptvChannels(new Set())
   314→      toast.success('M3U Parsed', { description: `Found ${result.total} channels` })
   315→    } catch {
   316→      toast.error('Error', { description: 'Failed to parse M3U file' })
   317→    } finally {
   318→      setIptvLoading(false)
   319→    }
   320→  }
   321→
   322→  const handleImportSelected = async () => {
   323→    setImporting(true)
   324→    let imported = 0
   325→    try {
   326→      for (const idx of selectedIptvChannels) {
   327→        const ch = iptvResults[idx]
   328→        if (ch) {
   329→          try {
   330→            await createChannel({
   331→              name: ch.name,
   332→              logo: ch.logo,
   333→              category: ch.group ? `sports,${ch.group.toLowerCase()}` : 'entertainment',
   334→              streamType: ch.url.includes('.m3u8') ? 'm3u' : 'iframe',
   335→              streamUrl: ch.url,
   336→            })
   337→            imported++
   338→          } catch {
   339→            // skip individual errors
   340→          }
   341→        }
   342→      }
   343→      toast.success('Import Complete', { description: `Successfully imported ${imported} channels` })
   344→      setShowIptvImport(false)
   345→      setIptvUrl('')
   346→      setIptvResults([])
   347→      setSelectedIptvChannels(new Set())
   348→      loadChannels()
   349→    } catch {
   350→      toast.error('Error', { description: 'Failed to import channels' })
   351→    } finally {
   352→      setImporting(false)
   353→    }
   354→  }
   355→
   356→  const toggleIptvChannel = (idx: number) => {
   357→    const next = new Set(selectedIptvChannels)
   358→    if (next.has(idx)) next.delete(idx)
   359→    else next.add(idx)
   360→    setSelectedIptvChannels(next)
   361→  }
   362→
   363→  const selectAllIptv = () => {
   364→    if (selectedIptvChannels.size === iptvResults.length) {
   365→      setSelectedIptvChannels(new Set())
   366→    } else {
   367→      setSelectedIptvChannels(new Set(iptvResults.map((_, i) => i)))
   368→    }
   369→  }
   370→
   371→  // ===== File Import Handlers =====
   372→
   373→  const handleFileUpload = async (file: File) => {
   374→    const ext = file.name.split('.').pop()?.toLowerCase()
   375→    if (ext !== 'm3u' && ext !== 'json') {
   376→      toast.error('Invalid File', { description: 'Only .m3u and .json files are supported' })
   377→      return
   378→    }
   379→
   380→    setFileImportLoading(true)
   381→    setFileImportResults([])
   382→    setSelectedFileChannels(new Set())
   383→
   384→    try {
   385→      const content = await file.text()
   386→      const fileType = ext === 'm3u' ? 'm3u' : 'json'
   387→      const result = await importFileContent(content, fileType)
   388→      setFileImportResults(result.channels)
   389→      toast.success('File Parsed', { description: `Found ${result.total} channels in ${file.name}` })
   390→    } catch (err) {
   391→      const msg = err instanceof Error ? err.message : 'Failed to parse file'
   392→      toast.error('Parse Error', { description: msg })
   393→    } finally {
   394→      setFileImportLoading(false)
   395→    }
   396→  }
   397→
   398→  const handleFileDrop = (e: React.DragEvent) => {
   399→    e.preventDefault()
   400→    setFileDragOver(false)
   401→    const file = e.dataTransfer.files[0]
   402→    if (file) handleFileUpload(file)
   403→  }
   404→
   405→  const handleFileInputChange = (e: React.ChangeEvent<HTMLInputElement>) => {
   406→    const file = e.target.files?.[0]
   407→    if (file) handleFileUpload(file)
   408→    // Reset input so same file can be re-selected
   409→    e.target.value = ''
   410→  }
   411→
   412→  const toggleFileChannel = (idx: number) => {
   413→    const next = new Set(selectedFileChannels)
   414→    if (next.has(idx)) next.delete(idx)
   415→    else next.add(idx)
   416→    setSelectedFileChannels(next)
   417→  }
   418→
   419→  const selectAllFileChannels = () => {
   420→    if (selectedFileChannels.size === fileImportResults.length) {
   421→      setSelectedFileChannels(new Set())
   422→    } else {
   423→      setSelectedFileChannels(new Set(fileImportResults.map((_, i) => i)))
   424→    }
   425→  }
   426→
   427→  const handleFileImportSelected = async () => {
   428→    setFileImporting(true)
   429→    let imported = 0
   430→    try {
   431→      for (const idx of selectedFileChannels) {
   432→        const ch = fileImportResults[idx]
   433→        if (ch) {
   434→          try {
   435→            await createChannel({
   436→              name: ch.name,
   437→              logo: ch.logo,
   438→              category: ch.group || 'entertainment',
   439→              streamType: ch.url.includes('.m3u8') ? 'm3u' : ch.url.includes('.ts') ? 'mpegts' : ch.url ? 'iframe' : 'm3u',
   440→              streamUrl: ch.url,
   441→              language: ch.language || '',
   442→              country: ch.country || '',
   443→            })
   444→            imported++
   445→          } catch {
   446→            // skip individual errors
   447→          }
   448→        }
   449→      }
   450→      toast.success('Import Complete', { description: `Successfully imported ${imported} channels` })
   451→      setShowFileImport(false)
   452→      setFileImportResults([])
   453→      setSelectedFileChannels(new Set())
   454→      loadChannels()
   455→    } catch {
   456→      toast.error('Error', { description: 'Failed to import channels' })
   457→    } finally {
   458→      setFileImporting(false)
   459→    }
   460→  }
   461→
   462→  return (

   464→      {/* Header bar */}
   465→      <div className="flex items-center justify-between gap-3 flex-wrap">
   466→        <div className="flex items-center gap-2 flex-1 min-w-0">
   467→          <div className="relative flex-1 max-w-sm">
   468→            <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
   469→            <Input
   470→              placeholder="Search channels..."
   471→              value={searchQuery}
   472→              onChange={(e) => setSearchQuery(e.target.value)}
   473→              className="pl-9 h-9"
   474→            />
   475→          </div>
   476→          <select
   477→            value={filterCategory}
   478→            onChange={(e) => setFilterCategory(e.target.value)}
   479→            className="h-9 rounded-lg border border-input bg-background px-3 text-sm"
   480→          >
   481→            <option value="all">All Categories</option>
   482→            {categoryOptions.map(opt => (
   483→              <option key={opt.value} value={opt.value}>{opt.label}</option>
   484→            ))}
   485→          </select>
   486→        </div>
   487→        <div className="flex gap-2 flex-wrap">
   488→          <Button
   489→            variant="outline"
   490→            size="sm"
   491→            onClick={() => { setShowIptvImport(!showIptvImport); setShowFileImport(false) }}
   492→            className="gap-1.5 btn-press text-xs h-9"
   493→          >
   494→            <Github className="h-3.5 w-3.5" />
   495→            <span className="hidden sm:inline">IPTV Import</span>
   496→            <span className="sm:hidden">IPTV</span>
   497→          </Button>
   498→          <Button
   499→            variant="outline"
   500→            size="sm"
   501→            onClick={() => { setShowFileImport(!showFileImport); setShowIptvImport(false) }}
   502→            className="gap-1.5 btn-press text-xs h-9"
   503→          >
   504→            <FileUp className="h-3.5 w-3.5" />
   505→            <span className="hidden sm:inline">File Import</span>
   506→            <span className="sm:hidden">File</span>
   507→          </Button>
   508→          <Button
   509→            variant="outline"
   510→            size="sm"
   511→            onClick={loadChannels}
   512→            className="gap-1.5 btn-press h-9"
   513→          >
   514→            <RefreshCw className={`h-3.5 w-3.5 ${loading ? 'animate-spin' : ''}`} />
   515→          </Button>
   516→          <Button
   517→            size="sm"
   518→            onClick={() => {
   519→              setEditingId(null)
   520→              setForm(emptyForm)
   521→              setShowForm(!showForm)
   522→            }}
   523→            className="gap-1.5 btn-press text-xs h-9"
   524→          >
   525→            <Plus className="h-3.5 w-3.5" />
   526→            Add Channel
   527→          </Button>
   528→        </div>
   529→      </div>
   530→
   531→      {/* IPTV Import Section */}
   532→      {showIptvImport && (
   533→        <div className="bg-card rounded-xl border border-border shadow-sm p-5 space-y-3">
   534→          <div className="flex items-center gap-2">
   535→            <Github className="h-4 w-4 text-primary" />
   536→            <h3 className="text-sm font-semibold">Import from GitHub M3U</h3>
   537→          </div>
   538→          <p className="text-xs text-muted-foreground">
   539→            Paste a GitHub raw M3U file URL. The system will parse and extract all channels.
   540→          </p>
   541→          <div className="flex gap-2">
   542→            <Input
   543→              placeholder="https://raw.githubusercontent.com/.../playlist.m3u"
   544→              value={iptvUrl}
   545→              onChange={(e) => setIptvUrl(e.target.value)}
   546→              className="flex-1 h-9"
   547→            />
   548→            <Button onClick={handleIptvParse} disabled={iptvLoading} size="sm" className="btn-press gap-1.5">
   549→              {iptvLoading ? (
   550→                <RefreshCw className="h-3.5 w-3.5 animate-spin" />
   551→              ) : (
   552→                <Upload className="h-3.5 w-3.5" />
   553→              )}
   554→              {iptvLoading ? 'Parsing...' : 'Parse'}
   555→            </Button>
   556→          </div>
   557→
   558→          {/* Parsed Results */}
   559→          {iptvResults.length > 0 && (
   560→            <div className="space-y-3">
   561→              {/* Search bar for parsed channels */}
   562→              <div className="relative">
   563→                <Search className="absolute left-2.5 top-1/2 -translate-y-1/2 h-3.5 w-3.5 text-muted-foreground" />
   564→                <Input
