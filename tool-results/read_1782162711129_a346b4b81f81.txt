    43→// Detect raw MPEG-TS stream URLs (.ts extension, not inside m3u8)
    44→function isTsUrl(url: string): boolean {
    45→  if (!url) return false
    46→  try {
    47→    const pathname = new URL(url).pathname
    48→    // Match .ts extension but NOT .m3u8 or playlist files
    49→    return /\.ts(\?.*)?$/.test(pathname) && !pathname.includes('.m3u8')
    50→  } catch {
    51→    // If URL parsing fails, do a simple check
    52→    return /\.ts(\?|$)/.test(url) && !url.includes('.m3u8')
    53→  }
    54→}

    88→// Route redirect URLs through our iframe proxy (used for 'redirect' type which always proxies)
    89→function proxyIframeUrl(url: string): string {
    90→  if (!url) return url
    91→  return `/api/iframe-proxy?url=${encodeURIComponent(url)}`

    94→// Note: iframe streams are routed through /api/iframe-proxy by the IframePlayer
    95→// component itself (it injects an ad-blocker and strips X-Frame-Options).
    96→// video-player passes the RAW iframe URL; IframePlayer does the proxying.
    97→
    98→// Route all streams through Next.js stream-proxy.
    99→// Previously, live .ts streams used a dedicated mini-service on port 3031,
   100→// but that caused dependency issues when the service wasn't running.
   101→// The Next.js stream-proxy handles all stream types reliably.
   102→function proxyStreamUrl(url: string, type: string): string {
   103→  if (!url) return url
   104→  // All streams (mpegts, m3u8, direct) → Next.js stream-proxy
   105→  return `/api/stream-proxy?url=${encodeURIComponent(url)}`
   106→}

   108→// Compute initial resolved URL & type synchronously to avoid flash of wrong URL
   109→// IMPORTANT: .ts URL detection must come BEFORE type-based routing, because
   110→// the stream type in the database might be incorrectly set (e.g. 'iframe' for a .ts URL)
   111→function getInitialResolved(url: string, type: string): { resolvedUrl: string; resolvedType: string } {
   112→  if (!url) return { resolvedUrl: url, resolvedType: type }
   113→  // Check for .ts URLs first — regardless of what type is stored
   114→  if (type === 'mpegts' || isTsUrl(url)) return { resolvedUrl: proxyStreamUrl(url, 'mpegts'), resolvedType: 'mpegts' }
   115→  if (type === 'redirect') return { resolvedUrl: url, resolvedType: 'iframe' } // IframePlayer proxies it
   116→  if (type === 'iframe') return { resolvedUrl: url, resolvedType: 'iframe' }
   117→  if (type === 'github_m3u') return { resolvedUrl: url, resolvedType: type } // resolved async
   118→  if (type === 'm3u8_jw') return { resolvedUrl: url, resolvedType: 'm3u8_jw' } // JW Player handles its own proxy
   119→  if (type === 'direct' || type === 'm3u' || type === 'm3u8') {
   120→    // Return the ORIGINAL URL directly — the HlsPlayer will try direct first,
   121→    // then fall back to proxy (via the proxyUrl prop) if direct fails or times out
   122→    return { resolvedUrl: url, resolvedType: type === 'direct' ? 'm3u' : type }
   123→  }
   124→  return { resolvedUrl: url, resolvedType: type }
   125→}

   250→  // Resolve stream URLs — detect .ts files and apply CORS proxy
   251→  //
   252→  // IMPORTANT: this effect re-runs whenever the user switches to a different
   253→  // stream channel (the `streamUrl` / `streamType` props change). When that
   254→  // happens we MUST clear the previous stream's transient player state —
   255→  // most importantly the `error` message — so that a stale error from the
   256→  // previous stream doesn't keep covering the new stream's playback area.
   257→  // (User complaint: switching streams after an error kept the old error
   258→  // message on screen.)
   259→  useEffect(() => {
   260→    async function resolve() {
   261→      // Reset all transient state so the new stream starts fresh.
   262→      setError(null)
   263→      setLoading(true)
   264→      setBuffering(false)
   265→      setQualityLevels([])
   266→      setHlsStats(null)
   267→      setCurrentQuality(-1)

   271→      setIsBehindLive(false)

   273→      setCurrentAudioTrack(-1)
   274→      setSubtitleTracks([])
   275→      setCurrentSubtitleTrack(-1)

   278→        try {
   279→          let url = streamUrl
   280→          if (url.includes('github.com') && !url.includes('raw.githubusercontent.com')) {
   281→            url = url.replace('github.com', 'raw.githubusercontent.com').replace('/blob/', '/')
   282→          }
   283→          const res = await fetch('/api/m3u-parse', {
   284→            method: 'POST',
   285→            headers: { 'Content-Type': 'application/json' },
   286→            body: JSON.stringify({ url }),
   287→          })
   288→          const data = await res.json()
   289→          if (data.channels && data.channels.length > 0) {
   290→            const channelUrl = data.channels[0].url
   291→            // Detect if resolved URL is a .ts stream
   292→            if (isTsUrl(channelUrl)) {
   293→              setResolvedUrl(`/api/stream-proxy?url=${encodeURIComponent(channelUrl)}`)
   294→              setResolvedType('mpegts')
   295→            } else {
   296→              setResolvedUrl(channelUrl)
   297→              setResolvedType('m3u')
   298→            }
   299→            onStreamResolved?.(channelUrl)
   300→          } else {
   301→            setError('No streams found in M3U file')
   302→          }
   303→        } catch {
   304→          setError('Failed to parse M3U file')
   305→        } finally {
   306→          setLoading(false)
   307→        }
   308→      } else if (streamType === 'redirect' && streamUrl) {
   309→        // Treat redirect as iframe — IframePlayer will proxy the URL
   310→        setResolvedType('iframe')
   311→        setResolvedUrl(streamUrl)
   312→      } else {
   313→        // Detect .ts URLs for MPEG-TS player (or explicit mpegts type)
   314→        if (streamType === 'mpegts' || isTsUrl(streamUrl)) {
   315→          setResolvedUrl(proxyStreamUrl(streamUrl, 'mpegts'))
   316→          setResolvedType('mpegts')
   317→        } else if (streamType === 'direct' || streamType === 'm3u' || streamType === 'm3u8') {
   318→          // Use the ORIGINAL URL directly — HlsPlayer tries direct first,
   319→          // then falls back to proxy via the proxyUrl prop
   320→          setResolvedUrl(streamUrl)
   321→          setResolvedType(streamType === 'direct' ? 'm3u' : streamType)
   322→        } else if (streamType === 'iframe' && streamUrl) {
   323→          // Pass the RAW URL — IframePlayer routes it through the proxy itself
   324→          // (the proxy injects an ad-blocker and strips X-Frame-Options).
   325→          setResolvedUrl(streamUrl)
   326→          setResolvedType('iframe')
   327→        } else {
   328→          setResolvedUrl(streamUrl)
   329→          setResolvedType(streamType)
   330→        }
   331→      }
   332→    }
   333→    resolve()
   334→  }, [streamUrl, streamType, onStreamResolved])
   335→
   336→  // Show controls with auto-hide timer
   337→  const showControls = useCallback(() => {

   339→    // Also show cursor in fullscreen when controls appear
   340→    if (fullscreen) {

   342→      if (cursorHideTimerRef.current) clearTimeout(cursorHideTimerRef.current)
   343→      cursorHideTimerRef.current = setTimeout(() => {
   344→        setCursorVisible(false)
   345→      }, 3000)
   346→    }
   347→    if (hideTimerRef.current) clearTimeout(hideTimerRef.current)
   348→    const isMobile = typeof window !== 'undefined' && 'ontouchstart' in window
   349→    const hideDelay = isIframe ? (isMobile ? 2500 : 3000) : 3000
   350→    hideTimerRef.current = setTimeout(() => {
   351→      if ((isIframe || playing) && !controlsBusy) setControlsVisible(false)
   352→    }, hideDelay)
   353→  }, [playing, controlsBusy, isIframe, fullscreen])
   354→
   355→  const toggleControlsVisibility = useCallback(() => {
   356→    if (controlsVisible) {
   357→      setControlsVisible(false)
   358→      if (hideTimerRef.current) clearTimeout(hideTimerRef.current)
   359→    } else {
   360→      showControls()
   361→    }
   362→  }, [controlsVisible, showControls])
   363→
   364→  const handleMouseMove = useCallback(() => {
   365→    showControls()
   366→    // Show cursor in fullscreen mode, then hide after inactivity
   367→    if (fullscreen) {
   368→      setCursorVisible(true)
   369→      if (cursorHideTimerRef.current) clearTimeout(cursorHideTimerRef.current)
   370→      cursorHideTimerRef.current = setTimeout(() => {
   371→        setCursorVisible(false)
   372→      }, 3000)
   373→    }

   375→
   376→  // ── Native touch event listeners for HLS swipe gestures (passive: false to prevent scroll) ──
   377→  // React's synthetic onTouchMove is passive by default — e.preventDefault() doesn't work.
   378→  // We must use native DOM listeners with { passive: false }.
   379→  useEffect(() => {
   380→    const el = containerRef.current
   381→    if (!el || isIframe) return
   382→
   383→    const handleTouchStart = (e: TouchEvent) => {
   384→      const touch = e.touches[0]
   385→      const rect = el.getBoundingClientRect()
   386→      if (!rect) return
   387→
   388→      const x = touch.clientX - rect.left
   389→      const halfWidth = rect.width / 2
   390→      const side = x < halfWidth ? 'left' : 'right'
   391→
   392→      gestureRef.current = {
   393→        startX: touch.clientX,
   394→        startY: touch.clientY,
   395→        startTime: Date.now(),
   396→        isSwiping: false,
   397→        side,
   398→        currentBrightness: brightnessRef.current,
   399→        startVolume: volumeRef.current,
   400→      }
   401→    }
   402→
   403→    const handleTouchMove = (e: TouchEvent) => {
   404→      if (!gestureRef.current) return
   405→      const touch = e.touches[0]
   406→      const deltaY = gestureRef.current.startY - touch.clientY // Positive = swipe up
   407→      const deltaX = Math.abs(touch.clientX - gestureRef.current.startX)
   408→
   409→      // Only activate swipe if vertical movement is dominant
   410→      if (!gestureRef.current.isSwiping) {
   411→        if (Math.abs(deltaY) > 20 && Math.abs(deltaY) > deltaX) {
   412→          gestureRef.current.isSwiping = true
   413→        } else {
   414→          return
   415→        }
   416→      }
   417→
   418→      // ★ CRITICAL: Prevent page scroll while swiping — only works with passive:false
   419→      e.preventDefault()
   420→
   421→      const containerHeight = el.getBoundingClientRect().height || 400
   422→
   423→      if (gestureRef.current.side === 'right') {
   424→        // Right side = volume control
   425→        const volumeChange = deltaY / containerHeight
   426→        const finalVolume = Math.max(0, Math.min(1, gestureRef.current.startVolume + volumeChange * 1.5))
   427→
   428→        setVolume(finalVolume)
   429→        if (mutedRef.current && finalVolume > 0) setMuted(false)
   430→
   431→        if (gestureIndicatorTimer.current) clearTimeout(gestureIndicatorTimer.current)
   432→        setGestureIndicator({ type: 'volume', value: finalVolume, visible: true })
   433→        gestureIndicatorTimer.current = setTimeout(() => {
   434→          setGestureIndicator(null)
   435→        }, 800)
   436→      } else {
   437→        // Left side = brightness control
   438→        const brightnessChange = deltaY / containerHeight
   439→        const newBrightness = Math.max(0.1, Math.min(1.5, gestureRef.current.currentBrightness + brightnessChange * 1.5))
   440→        brightnessRef.current = newBrightness
   441→
   442→        const videoEl = el.querySelector('video')
   443→        if (videoEl) {
   444→          videoEl.style.filter = `brightness(${newBrightness})`
   445→        }
   446→
   447→        if (gestureIndicatorTimer.current) clearTimeout(gestureIndicatorTimer.current)
   448→        setGestureIndicator({ type: 'brightness', value: newBrightness, visible: true })
   449→        gestureIndicatorTimer.current = setTimeout(() => {
   450→          setGestureIndicator(null)
   451→        }, 800)
   452→      }
   453→    }
   454→
   455→    const handleTouchEnd = () => {
   456→      gestureRef.current = null
   457→    }
   458→
   459→    // Register with passive:false so preventDefault() works
   460→    el.addEventListener('touchstart', handleTouchStart, { passive: true })
   461→    el.addEventListener('touchmove', handleTouchMove, { passive: false })
   462→    el.addEventListener('touchend', handleTouchEnd, { passive: true })
   463→
   464→    return () => {
   465→      el.removeEventListener('touchstart', handleTouchStart)
   466→      el.removeEventListener('touchmove', handleTouchMove)
   467→      el.removeEventListener('touchend', handleTouchEnd)
   468→    }
   469→  }, [isIframe]) // Re-attach when player type changes
   470→
   471→  // Fullscreen change listener
   472→  useEffect(() => {
   473→    function handleFullscreenChange() {
   474→      const isFullscreen = !!document.fullscreenElement
   475→      setFullscreen(isFullscreen)
   476→      if (isFullscreen) {
   477→        // Entering fullscreen — start cursor hide timer
   478→        setCursorVisible(true)
   479→        if (cursorHideTimerRef.current) clearTimeout(cursorHideTimerRef.current)
   480→        cursorHideTimerRef.current = setTimeout(() => {
   481→          setCursorVisible(false)
   482→        }, 3000)
   483→      } else {
   484→        // Exiting fullscreen — show cursor and cancel timer
   485→        setCursorVisible(true)
   486→        if (cursorHideTimerRef.current) clearTimeout(cursorHideTimerRef.current)
   487→        try {
   488→          const screen = window.screen as ExtendedScreen
   489→          screen.orientation.unlock?.()
   490→        } catch {
   491→          // Orientation unlock not supported
   492→        }
   493→      }
   494→    }
   495→    document.addEventListener('fullscreenchange', handleFullscreenChange)
   496→    return () => document.removeEventListener('fullscreenchange', handleFullscreenChange)
   497→  }, [])
   498→
   499→  const togglePlay = useCallback(() => {
   500→    const video = videoRef.current || containerRef.current?.querySelector('video')
   501→    if (video) {
   502→      if (video.paused) {
   503→        video.play()
   504→        setPlaying(true)
   505→      } else {
   506→        video.pause()
   507→        setPlaying(false)
   508→      }
   509→    }
   510→  }, [])
   511→
   512→  const toggleFullscreen = useCallback(async () => {
   513→    if (!containerRef.current) return
   514→    if (document.fullscreenElement) {
   515→      try {
   516→        const screen = window.screen as ExtendedScreen
   517→        screen.orientation.unlock?.()
   518→      } catch {
   519→        // Orientation unlock not supported
   520→      }
   521→      document.exitFullscreen()
   522→    } else {
   523→      try {
   524→        await containerRef.current.requestFullscreen()
   525→        try {
   526→          const screen = window.screen as ExtendedScreen
   527→          if (screen.orientation?.lock) {
   528→            await screen.orientation.lock('landscape')
