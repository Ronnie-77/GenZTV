    45→// Detect raw MPEG-TS stream URLs (.ts extension, not inside m3u8)
    46→function isTsUrl(url: string): boolean {
    47→  if (!url) return false
    48→  try {
    49→    const pathname = new URL(url).pathname
    50→    // Match .ts extension but NOT .m3u8 or playlist files
    51→    return /\.ts(\?.*)?$/.test(pathname) && !pathname.includes('.m3u8')
    52→  } catch {
    53→    // If URL parsing fails, do a simple check
    54→    return /\.ts(\?|$)/.test(url) && !url.includes('.m3u8')
    55→  }
    56→}

    96→// Note: iframe streams are routed through /api/iframe-proxy by the IframePlayer
    97→// component itself (it injects an ad-blocker and strips X-Frame-Options).
    98→// video-player passes the RAW iframe URL; IframePlayer does the proxying.
    99→
   100→// Route all streams through Next.js stream-proxy.
   101→// Previously, live .ts streams used a dedicated mini-service on port 3031,
   102→// but that caused dependency issues when the service wasn't running.
   103→// The Next.js stream-proxy handles all stream types reliably.
   104→function proxyStreamUrl(url: string, type: string): string {
   105→  if (!url) return url
   106→  // All streams (mpegts, m3u8, direct) → Next.js stream-proxy
   107→  return `/api/stream-proxy?url=${encodeURIComponent(url)}`
   108→}

   110→// Compute initial resolved URL & type synchronously to avoid flash of wrong URL
   111→// IMPORTANT: .ts URL detection must come BEFORE type-based routing, because
   112→// the stream type in the database might be incorrectly set (e.g. 'iframe' for a .ts URL)
   113→function getInitialResolved(url: string, type: string): { resolvedUrl: string; resolvedType: string } {
   114→  if (!url) return { resolvedUrl: url, resolvedType: type }
   115→  // Check for .ts URLs first — regardless of what type is stored
   116→  if (type === 'mpegts' || isTsUrl(url)) return { resolvedUrl: proxyStreamUrl(url, 'mpegts'), resolvedType: 'mpegts' }
   117→  if (type === 'redirect') return { resolvedUrl: url, resolvedType: 'iframe' } // IframePlayer proxies it
   118→  if (type === 'iframe') return { resolvedUrl: url, resolvedType: 'iframe' }
   119→  if (type === 'github_m3u') return { resolvedUrl: url, resolvedType: type } // resolved async
   120→  if (type === 'm3u8_jw') return { resolvedUrl: url, resolvedType: 'm3u8_jw' } // JW Player handles its own proxy
   121→  // 🎯 Dedicated Direct HLS player — CORS-open CDNs, low-latency optimized
   122→  if (type === 'm3u8_direct') return { resolvedUrl: url, resolvedType: 'm3u8_direct' }
   123→  // 🛡️ Dedicated Proxy HLS player — always via /api/stream-proxy, CORS/Referer bypass
   124→  if (type === 'm3u8_proxy') return { resolvedUrl: url, resolvedType: 'm3u8_proxy' }
   125→  if (type === 'direct' || type === 'm3u' || type === 'm3u8') {
   126→    // Return the ORIGINAL URL directly — the HlsPlayer will try direct first,
   127→    // then fall back to proxy (via the proxyUrl prop) if direct fails or times out
   128→    return { resolvedUrl: url, resolvedType: type === 'direct' ? 'm3u' : type }
   129→  }
   130→  return { resolvedUrl: url, resolvedType: type }
   131→}

   261→  isIframeRef.current = isIframe
   262→
   263→  // Resolve stream URLs — detect .ts files and apply CORS proxy
   264→  //
   265→  // IMPORTANT: this effect re-runs whenever the user switches to a different
   266→  // stream channel (the `streamUrl` / `streamType` props change). When that
   267→  // happens we MUST clear the previous stream's transient player state —
   268→  // most importantly the `error` message — so that a stale error from the
   269→  // previous stream doesn't keep covering the new stream's playback area.
   270→  // (User complaint: switching streams after an error kept the old error
   271→  // message on screen.)
   272→  useEffect(() => {
   273→    async function resolve() {
   274→      // Reset all transient state so the new stream starts fresh.
   275→      setError(null)
   276→      setLoading(true)
   277→      setBuffering(false)
   278→      setQualityLevels([])
   279→      setHlsStats(null)
   280→      setCurrentQuality(-1)
   281→      setHlsLoadMode('direct')
   282→      setHlsFallbackMpegts(false)
   283→      setPlaying(false)
   284→      setIsBehindLive(false)
   285→      setAudioTracks([])
   286→      setCurrentAudioTrack(-1)
   287→      setSubtitleTracks([])
   288→      setCurrentSubtitleTrack(-1)
   289→
   290→      if (streamType === 'github_m3u' && streamUrl) {
   291→        try {
   292→          let url = streamUrl
   293→          if (url.includes('github.com') && !url.includes('raw.githubusercontent.com')) {
   294→            url = url.replace('github.com', 'raw.githubusercontent.com').replace('/blob/', '/')
   295→          }
   296→          const res = await fetch('/api/m3u-parse', {
   297→            method: 'POST',
   298→            headers: { 'Content-Type': 'application/json' },
   299→            body: JSON.stringify({ url }),
   300→          })
   301→          const data = await res.json()
   302→          if (data.channels && data.channels.length > 0) {
   303→            const channelUrl = data.channels[0].url
   304→            // Detect if resolved URL is a .ts stream
   305→            if (isTsUrl(channelUrl)) {
   306→              setResolvedUrl(`/api/stream-proxy?url=${encodeURIComponent(channelUrl)}`)
   307→              setResolvedType('mpegts')
   308→            } else {
   309→              setResolvedUrl(channelUrl)
   310→              setResolvedType('m3u')
   311→            }
   312→            onStreamResolved?.(channelUrl)
   313→          } else {
   314→            setError('No streams found in M3U file')
   315→          }
   316→        } catch {
   317→          setError('Failed to parse M3U file')
   318→        } finally {
   319→          setLoading(false)
   320→        }
   321→      } else if (streamType === 'redirect' && streamUrl) {
   322→        // Treat redirect as iframe — IframePlayer will proxy the URL
   323→        setResolvedType('iframe')
   324→        setResolvedUrl(streamUrl)
   325→      } else {
   326→        // Detect .ts URLs for MPEG-TS player (or explicit mpegts type)
   327→        if (streamType === 'mpegts' || isTsUrl(streamUrl)) {
   328→          setResolvedUrl(proxyStreamUrl(streamUrl, 'mpegts'))
   329→          setResolvedType('mpegts')
   330→        } else if (streamType === 'm3u8_direct') {
   331→          // 🎯 Dedicated direct player — pass raw URL, player handles everything
   332→          setResolvedUrl(streamUrl)
   333→          setResolvedType('m3u8_direct')
   334→        } else if (streamType === 'm3u8_proxy') {
   335→          // 🛡️ Dedicated proxy player — pass raw URL, player wraps it in /api/stream-proxy
   336→          setResolvedUrl(streamUrl)
   337→          setResolvedType('m3u8_proxy')
   338→        } else if (streamType === 'direct' || streamType === 'm3u' || streamType === 'm3u8') {
   339→          // Use the ORIGINAL URL directly — HlsPlayer tries direct first,
   340→          // then falls back to proxy via the proxyUrl prop
   341→          setResolvedUrl(streamUrl)
   342→          setResolvedType(streamType === 'direct' ? 'm3u' : streamType)
   343→        } else if (streamType === 'iframe' && streamUrl) {
   344→          // Pass the RAW URL — IframePlayer routes it through the proxy itself
   345→          // (the proxy injects an ad-blocker and strips X-Frame-Options).
   346→          setResolvedUrl(streamUrl)
   347→          setResolvedType('iframe')
   348→        } else {
   349→          setResolvedUrl(streamUrl)
   350→          setResolvedType(streamType)
   351→        }
   352→      }
   353→    }
   354→    resolve()
   355→  }, [streamUrl, streamType, onStreamResolved])
   356→

   366→      }, 3000)

   369→    const isMobile = typeof window !== 'undefined' && 'ontouchstart' in window
   370→    const hideDelay = isIframe ? (isMobile ? 2500 : 3000) : 3000
   371→    hideTimerRef.current = setTimeout(() => {
   372→      if ((isIframe || playing) && !controlsBusy) setControlsVisible(false)
   373→    }, hideDelay)
   374→  }, [playing, controlsBusy, isIframe, fullscreen])
   375→
   376→  const toggleControlsVisibility = useCallback(() => {
   377→    if (controlsVisible) {
   378→      setControlsVisible(false)
   379→      if (hideTimerRef.current) clearTimeout(hideTimerRef.current)
   380→    } else {
   381→      showControls()
   382→    }
   383→  }, [controlsVisible, showControls])
   384→
   385→  const handleMouseMove = useCallback(() => {
   386→    showControls()
   387→    // Show cursor in fullscreen mode, then hide after inactivity
   388→    if (fullscreen) {
   389→      setCursorVisible(true)
   390→      if (cursorHideTimerRef.current) clearTimeout(cursorHideTimerRef.current)
   391→      cursorHideTimerRef.current = setTimeout(() => {
   392→        setCursorVisible(false)
   393→      }, 3000)
   394→    }
   395→  }, [showControls, fullscreen])
   396→
   397→  // ── Native touch event listeners for HLS swipe gestures (passive: false to prevent scroll) ──
   398→  // React's synthetic onTouchMove is passive by default — e.preventDefault() doesn't work.
   399→  // We must use native DOM listeners with { passive: false }.
   400→  useEffect(() => {
   401→    const el = containerRef.current
   402→    if (!el || isIframe) return
   403→
   404→    const handleTouchStart = (e: TouchEvent) => {
   405→      const touch = e.touches[0]
   406→      const rect = el.getBoundingClientRect()
   407→      if (!rect) return
   408→
   409→      const x = touch.clientX - rect.left
   410→      const halfWidth = rect.width / 2
   411→      const side = x < halfWidth ? 'left' : 'right'
   412→
   413→      gestureRef.current = {
   414→        startX: touch.clientX,
   415→        startY: touch.clientY,
   416→        startTime: Date.now(),
   417→        isSwiping: false,
   418→        side,
   419→        currentBrightness: brightnessRef.current,
   420→        startVolume: volumeRef.current,
   421→      }
   422→    }
   423→
   424→    const handleTouchMove = (e: TouchEvent) => {
   425→      if (!gestureRef.current) return
   426→      const touch = e.touches[0]
   427→      const deltaY = gestureRef.current.startY - touch.clientY // Positive = swipe up
   428→      const deltaX = Math.abs(touch.clientX - gestureRef.current.startX)
   429→
   430→      // Only activate swipe if vertical movement is dominant
   431→      if (!gestureRef.current.isSwiping) {
   432→        if (Math.abs(deltaY) > 20 && Math.abs(deltaY) > deltaX) {
   433→          gestureRef.current.isSwiping = true
   434→        } else {
   435→          return
   436→        }
   437→      }
   438→
   439→      // ★ CRITICAL: Prevent page scroll while swiping — only works with passive:false
   440→      e.preventDefault()
   441→
   442→      const containerHeight = el.getBoundingClientRect().height || 400
   443→
   444→      if (gestureRef.current.side === 'right') {
   445→        // Right side = volume control
   446→        const volumeChange = deltaY / containerHeight
   447→        const finalVolume = Math.max(0, Math.min(1, gestureRef.current.startVolume + volumeChange * 1.5))
   448→
   449→        setVolume(finalVolume)
   450→        if (mutedRef.current && finalVolume > 0) setMuted(false)
   451→
   452→        if (gestureIndicatorTimer.current) clearTimeout(gestureIndicatorTimer.current)
   453→        setGestureIndicator({ type: 'volume', value: finalVolume, visible: true })
   454→        gestureIndicatorTimer.current = setTimeout(() => {
   455→          setGestureIndicator(null)
   456→        }, 800)
   457→      } else {
   458→        // Left side = brightness control
   459→        const brightnessChange = deltaY / containerHeight
   460→        const newBrightness = Math.max(0.1, Math.min(1.5, gestureRef.current.currentBrightness + brightnessChange * 1.5))
   461→        brightnessRef.current = newBrightness
   462→
   463→        const videoEl = el.querySelector('video')
   464→        if (videoEl) {
   465→          videoEl.style.filter = `brightness(${newBrightness})`
   466→        }
   467→
   468→        if (gestureIndicatorTimer.current) clearTimeout(gestureIndicatorTimer.current)
   469→        setGestureIndicator({ type: 'brightness', value: newBrightness, visible: true })
   470→        gestureIndicatorTimer.current = setTimeout(() => {
   471→          setGestureIndicator(null)
   472→        }, 800)
   473→      }
   474→    }
   475→
   476→    const handleTouchEnd = () => {
   477→      gestureRef.current = null
   478→    }
   479→
   480→    // Register with passive:false so preventDefault() works
   481→    el.addEventListener('touchstart', handleTouchStart, { passive: true })
   482→    el.addEventListener('touchmove', handleTouchMove, { passive: false })
   483→    el.addEventListener('touchend', handleTouchEnd, { passive: true })
   484→
   485→    return () => {
   486→      el.removeEventListener('touchstart', handleTouchStart)
   487→      el.removeEventListener('touchmove', handleTouchMove)
   488→      el.removeEventListener('touchend', handleTouchEnd)
   489→    }
   490→  }, [isIframe]) // Re-attach when player type changes
   491→
   492→  // Fullscreen change listener
   493→  useEffect(() => {
   494→    function handleFullscreenChange() {
   495→      const isFullscreen = !!document.fullscreenElement
   496→      setFullscreen(isFullscreen)
   497→      if (isFullscreen) {
   498→        // Entering fullscreen — start cursor hide timer
   499→        setCursorVisible(true)
   500→        if (cursorHideTimerRef.current) clearTimeout(cursorHideTimerRef.current)
   501→        cursorHideTimerRef.current = setTimeout(() => {
   502→          setCursorVisible(false)
